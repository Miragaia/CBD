// // 21.Apresente o número total de avaliações (numGrades) em cada dia da semana.

// Input:

db.restaurants.aggregate([
  {
    $unwind: "$grades"
  },
  {
    $project: {
      dayOfWeek: { $dayOfWeek: "$grades.date" }
    }
  },
  {
    $group: {
      _id: "$dayOfWeek",
      numGrades: { $sum: 1 }
    }
  },
  {
    $sort: {
      _id: 1
    }
  },
  {
    $project: {
      _id: 0,
      dayOfWeek: "$_id",
      numGrades: 1
    }
  }
])
